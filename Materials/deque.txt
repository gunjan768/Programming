#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,k,i,x,count;
    cin>>t;
    for(int m=1;m<=t;m++)
    {
        cin>>n>>k;
        deque<int>d1;
        for(i=0;i<n;i++)
        {
            cin>>x;
            d1.push_back(x);
        }
        for(i=0;i+k-1<d1.size();i++)
        {
            int max=d1[i];
            for(int j=i+1,count=0;count!=k-1;j++)
            {
                if(max<d1[j])
                  max=d1[j];
                count++;
            }
            cout<<max<<" ";
        }
         cout<<endl; 
    }
}
