void sieve()
{
    rep(i,2,10005)
    prime[i]=1;

    for(int i=2;i*i<10005;i++)
    {
        if(prime[i])
        for(ll j=i*i;j<10005;j+=i)
        prime[j]=0;
    }
} 

int min(int a,int b)
{
    if(a<b)
    return a;

    return b;
}

int manacher(string s1)
{
    n=s1.size();

    int palind[n+1]={};
    int r=0,c=0,maxLen=0;
  
    rep(i,0,n)
    {
        int mirror=2*c-i;

        if(i<r)
        palind[i]=min(r-i,palind[mirror]);

        int a=i+(palind[i]+1);
        int b=i-(palind[i]+1);

        while(a<n && b>=0 && s1[a]==s1[b])
        a++,b--,palind[i]++;

        if(i+palind[i]>r)
        {
            c=i;
            r=i+palind[i];

            maxLen=max(palind[i],maxLen);
        }
    }

    return maxLen;
}
