void multiply(ll f[2][2],ll m[2][2]){
    ll a=(f[0][0]*m[0][0]+f[0][1]*m[1][0])%mod;
    ll b=(f[0][0]*m[0][1]+f[0][1]*m[1][1])%mod;
    ll c=(f[1][0]*m[0][0]+f[1][1]*m[1][0])%mod;
    ll d=(f[1][0]*m[0][1]+f[1][1]*m[1][1])%mod;
    f[0][0]=a;  f[0][1]=b; 
    f[1][0]=c;  f[1][1]=d; 
}
void power(ll n,ll f[2][2]){
    if(!n or n==1) return ;
    ll m[2][2]={{1,1},{1,0}};
    power(n/2,f);
    multiply(f,f);
    if(n&1) multiply(f,m);
}
ll fibo(ll n){
    ll f[2][2]={{1,1},{1,0}};
    if(!n) return 0;
    power(n-1,f);
    f[0][0]%=mod;
    return f[0][0];
}
